Macro Example
 


Plot <font color=blue>Sine</font> and <font color=red>Cosine</font> Waves Example

<loop id=x min=0 max=360 increment=5>
<Store id=sin value="{x}:{eval(Math.sin({x.radian}) * 100 + 100)}"> 
<Store id=cos value="{x}:{eval(Math.cos({x.radian}) * 100 + 100)}"> 
</loop>
<br>

<graphic border=0 height=200 width=460 range="0,0,100,200">
Graph,fillcolor:white,0:0,360:200,,0,360,8,0,4,true,,-1,1,10,0,4,true;
Line,'Color:Blue;Weight:4',{sin.join};
Line,'Color:Red;Weight:4',{cos.join};
</graphic>
