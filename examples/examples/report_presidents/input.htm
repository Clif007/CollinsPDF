<SCRIPT LOAD>
function myfirst(value)
{
	var n,text,i;
	n = parseInt(value);
	if (isNaN(n)) return value;
	text = '' + n;
	i = text.substr(text.length-1);
	if (i == '1') return n + 'st';
	if (i == '2') return n + 'nd';
	if (i == '3') return n + 'th';
	return n + 'th';
}
</SCRIPT>
