<assign id=count value="35">

<random id=a min=50 max=99>
<random id=b min=50 max=99>
<random id=sign_a list=",-">
<random id=sign_b list=",-">
<random id=op list="+,-">

<group id=g count={count} columns=5 columnOrder=true >
<style> td {font-size:14pt} </style>

  <table cellpadding=0 cellspacing=0>
  <tr>
      <td width=20 align=right><font size=1>({g.count})</font></td>
      <td width=15></td>
      <td width=28 align=right>|{sign_a}|{a}</td>
  </tr>
  <tr>
      <td width=20></td>
      <td width=15><b>|{op}|</b></td>
      <td width=28 align=right><u> |{sign_b}|{b}</u></td>
  </tr>
  </table>

  <div style="height:70"></div>

<store id=answers value="{eval({sign_a.last}{a.last} {op.last} {sign_b.last}{b.last}}">
<br>
</group>
